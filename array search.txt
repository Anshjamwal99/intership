#include<iostream>
using namespace std;
int main(){
    int array[5]={10,20,30,40,50};
    int user_input;
    cout<<"enter the number u want to search"<<endl;
    cin>>user_input;
    int counter=0;
    for(auto x:array)
    {
        if(x==user_input)
        {
            cout<<"number"<<user_input<<"is found at "<<counter<<endl;
            exit(0);
            
        }
        counter++;
    }
  cout<<"not found";
  
    return 0;
}
