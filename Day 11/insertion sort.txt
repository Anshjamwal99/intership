#include <iostream>
#include <vector>
#include <algorithm>

// Function to perform insertion sort using STL functions
void insertionSort(std::vector<int> &arr) {
    int n = arr.size();
    for (int i = 1; i < n; ++i) {
        // Use std::find to locate the correct position to insert arr[i]
        auto it = std::find_if(arr.begin(), arr.begin() + i,
                               [&arr, &i](int element) { return element > arr[i]; });

        // Insert arr[i] in the correct position
        if (it != arr.begin() + i) {
            int temp = arr[i];
            arr.erase(arr.begin() + i);
            arr.insert(it, temp);
        }
    }
}

int main() {
    std::vector<int> arr = {12, 11, 13, 5, 6};
    
    std::cout << "Original array: ";
    for (int num : arr) {
        std::cout << num << " ";
    }
    std::cout << std::endl;

    insertionSort(arr);

    std::cout << "Sorted array: ";
    for (int num : arr) {
        std::cout << num << " ";
    }
    std::cout << std::endl;

    return 0;
}
