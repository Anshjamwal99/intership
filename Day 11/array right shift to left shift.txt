#include<iostream>
#include<bits/stdc++.h>
using namespace std;

void swaptofront(vector<int> &arrl){
    int temp=arrl[arrl.size()-1];
    for(int i=arrl.size()-1;i>0;i--){
        arrl[i]=arrl[i-1];
    }
    arrl[0]=temp;
}

int main(){
    vector<int> arr={1,2,3,4,5,6,7};
    for(int i=0;i<arr.size();i++){
    swaptofront(arr);
    for(auto i : arr){
        cout<<i<<" ";
    }
    cout<<endl;
    }
    return 0;
}
