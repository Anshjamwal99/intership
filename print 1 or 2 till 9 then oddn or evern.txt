#include <iostream>
using namespace std;

void printRangeDescription(int start, int end) {           void range(int start,int end)

    for (int num = start; num <= end; ++num) {                  for(int num=start;num<=end;++num){ 

                                                                 
        if (num == 1) {
            cout << "one" << endl;                          
        } else if (num == 2) {
            cout << "two" << endl;
        } else if (num == 3) {
            cout << "three" << endl;
        } else if (num == 4) {
            cout << "four" << endl;
        } else if (num == 5) {
            cout << "five" << endl;
        } else if (num == 6) {
            cout << "six" << endl;
        } else if (num == 7) {
            cout << "seven" << endl;
        } else if (num == 8) {
            cout << "eight" << endl;
        } else if (num == 9) {
            cout << "nine" << endl;
        } else {
            if (num % 2 == 0) {
                cout << "even" << endl;
            } else {
                cout << "odd" << endl;
            }
        }
    }
}

int main() {
    int start, end;
    cin >> start >> end;

    printRangeDescription(start, end);

    return 0;
}
