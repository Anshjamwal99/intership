#include <iostream>
#include <fstream>
#include <string>

using namespace std;

int main() {
    string filename = "new.txt";  // Replace with your file name
    ifstream file(filename);

    if (!file.is_open()) {
        cerr << "Error: Unable to open the file." << endl;
        return 1;
    }

    double number;
    double sum = 0.0;

    while (file >> number) {
        sum += number;
    }

    if (file.bad()) {
        cerr << "Error: An error occurred while reading the file." << endl;
        return 1;
    }

    file.close();
;
    cout << "Sum of numbers: " << sum << endl;

    return 0;
}
