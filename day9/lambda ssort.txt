#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
vector<int>  numbers={3,5,8,1};
sort(numbers.begin(),numbers.end(),[](int a,int b){
    return a<b;
});
cout<<"sorted numbers"<<endl;
for(int num:numbers){
    cout<<num<<" ";
}
cout<<endl;
    return 0;
}
